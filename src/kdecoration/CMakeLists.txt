project(bismuth-kdecoration)

add_library(bismuth_kdecoration MODULE)

target_sources(bismuth_kdecoration PRIVATE decoration.cpp decoration.hpp)

  PUBLIC Qt::Core
  PRIVATE KDecoration3::KDecoration KF6::ConfigCore KF6::ConfigWidgets KF6::CoreAddons)

        DESTINATION ${KDE_INSTALL_PLUGINDIR}/org.kde.kdecoration3)
